import math
import random

# ══════════════════════════════════════════════════════════════════════════════
# 1. NET DELTA EXPOSURE OF A PORTFOLIO
# ══════════════════════════════════════════════════════════════════════════════
# "Given a portfolio of options and stock positions, what is the net delta?"
#
# Assumptions I'd state out loud:
#   - positions is a list of dicts with keys: type, delta, quantity
#   - type is 'stock', 'call', or 'put'
#   - stock has delta = 1 by definition (moves <span class="nokatex">$</span>1 per <span class="nokatex">$</span>1 in spot)
#   - quantity is signed: positive = long, negative = short
#   - net delta tells us: for a <span class="nokatex">$</span>1 move in spot, how much does the portfolio move?

def net_delta(positions):
    total_delta = 0.0
    for pos in positions:
        if pos['type'] == 'stock':
            total_delta += pos['quantity'] * 1.0        # stock delta always 1
        else:
            total_delta += pos['quantity'] * pos['delta']
    return total_delta
    # if net delta > 0: portfolio is net long (profits when spot rises)
    # if net delta < 0: portfolio is net short (profits when spot falls)
    # if net delta ~ 0: delta-neutral (hedged against small spot moves)

positions = [
    {'type': 'stock',  'quantity':  100,  'delta': 1.0},   # long 100 shares
    {'type': 'call',   'quantity': -10,   'delta': 0.55},  # short 10 calls, delta 0.55
    {'type': 'put',    'quantity':  5,    'delta': -0.30},  # long 5 puts, delta -0.30
]

nd = net_delta(positions)
print(f"Net Delta: {nd:.2f}")
print(f"  -> portfolio moves ${nd:.2f} for every <span class="nokatex">$</span>1 move in spot")
print(f"  -> to delta-neutralize: {'sell' if nd > 0 else 'buy'} {abs(nd):.2f} shares\n")

# 2. PORTFOLIO VOLATILITY — two assets with correlation
# ══════════════════════════════════════════════════════════════════════════════
# "Given two assets with vols and a correlation, what is portfolio vol?"
#
# Assumptions I'd state out loud:
#   - weights sum to 1 (long-only), but formula works for any weights
#   - sigma_p = sqrt(w1^2*s1^2 + w2^2*s2^2 + 2*w1*w2*rho*s1*s2)
#   - if rho=1: no diversification, sigma_p = w1*s1 + w2*s2
#   - if rho=0: pure diversification, sigma_p < weighted average of vols
#   - if rho=-1: maximum diversification, can get sigma_p = 0

def portfolio_vol_2asset(w1, s1, w2, s2, rho):
    variance = (w1**2 * s1**2
                + w2**2 * s2**2
                + 2 * w1 * w2 * rho * s1 * s2)
    return math.sqrt(variance)
    # key insight for risk manager: correlation drives diversification benefit
    # in a crisis, correlations spike toward 1 -> diversification disappears

w1, s1 = 0.6, 0.20   # 60% in asset A, 20% annual vol
w2, s2 = 0.4, 0.15   # 40% in asset B, 15% annual vol
rho = 0.3             # moderate positive correlation

pvol = portfolio_vol_2asset(w1, s1, w2, s2, rho)
naive = w1*s1 + w2*s2  # weighted average, no diversification
print(f"Portfolio Vol (rho={rho}): {pvol:.2%}")
print(f"Naive weighted avg vol:   {naive:.2%}")
print(f"Diversification benefit:  {naive - pvol:.2%}\n")

# 3. MONTE CARLO OPTION PRICER
# ══════════════════════════════════════════════════════════════════════════════
# "Price a European call using Monte Carlo simulation"
#
# Assumptions I'd state out loud:
#   - log-normal stock price: S_T = S * exp((r - 0.5*sigma^2)*T + sigma*sqrt(T)*Z)
#     where Z ~ N(0,1) — this is the GBM (geometric Brownian motion) solution
#   - we simulate N terminal prices, compute payoff, discount back
#   - more paths = more accurate but slower; 100k is a good balance
#   - result will be close to BS price for a European call

def monte_carlo_call(S, K, T, r, sigma, n=100000):
    payoffs = []
    for _ in range(n):
        # Box-Muller: generate standard normal from two uniform randoms
        u1, u2 = random.random(), random.random()
        Z = math.sqrt(-2 * math.log(u1)) * math.cos(2 * math.pi * u2)

        # simulate terminal stock price under risk-neutral measure
        S_T = S * math.exp((r - 0.5 * sigma**2) * T + sigma * math.sqrt(T) * Z)

        # call payoff at expiry
        payoffs.append(max(S_T - K, 0))

    # discount expected payoff back to today
    price = math.exp(-r * T) * sum(payoffs) / n
    return price

mc_price = monte_carlo_call(S=45, K=50, T=1, r=0.05, sigma=0.25, n=200000)
print(f"Monte Carlo Call Price: ${mc_price:.4f}")
print(f"  (should match BS ~<span class="nokatex">$</span>4.08 for S=45, K=50, T=1, r=5%, sigma=25%)\n")

# 4. FULL RISK REPORT — "give me a risk summary for this return series"
# ══════════════════════════════════════════════════════════════════════════════
# "Write a function that takes a return series and outputs key risk metrics"
#
# Assumptions I'd state out loud:
#   - returns are daily
#   - Sharpe uses annualization factor sqrt(252)
#   - VaR is historical at 95%
#   - max drawdown converts returns to a price series starting at 100
#   - I'll return a dict so it's easy to read and extend

def risk_report(returns, risk_free_annual=0.0):
    n = len(returns)
    mean = sum(returns) / n
    variance = sum((r - mean)**2 for r in returns) / (n - 1)
    std = math.sqrt(variance)

    # annualized Sharpe
    daily_rf = risk_free_annual / 252
    excess = [r - daily_rf for r in returns]
    exc_mean = sum(excess) / n
    exc_std  = math.sqrt(sum((r - exc_mean)**2 for r in excess) / (n - 1))
    sharpe = (exc_mean / exc_std) * math.sqrt(252) if exc_std > 0 else None

    # historical VaR at 95%
    sorted_r = sorted(returns)
    var_idx  = int(0.05 * n)
    var_95   = sorted_r[var_idx]

    # CVaR (expected shortfall): average of losses beyond VaR
    cvar_95  = sum(sorted_r[:var_idx]) / var_idx if var_idx > 0 else sorted_r[0]

    # max drawdown — build price series from returns
    price = 100.0
    peak  = price
    max_dd = 0.0
    for r in returns:
        price *= (1 + r)
        if price > peak:
            peak = price
        dd = (price - peak) / peak
        if dd < max_dd:
            max_dd = dd

    return {
        'ann_return':  mean * 252,
        'ann_vol':     std * math.sqrt(252),
        'sharpe':      sharpe,
        'var_95':      var_95,
        'cvar_95':     cvar_95,
        'max_drawdown': max_dd,
    }

# realistic daily returns
random.seed(42)
returns = [random.gauss(0.0004, 0.012) for _ in range(252)]

report = risk_report(returns)
print("Risk Report:")
for k, v in report.items():
    print(f"  {k:15s}: {v:.4%}" if v is not None else f"  {k:15s}: N/A")
print()

# 5. CLEAN A MESSY RETURN SERIES
# ══════════════════════════════════════════════════════════════════════════════
# "Given a return series with bad data, clean it before computing risk metrics"
#
# Assumptions I'd state out loud:
#   - None values are missing data -> drop them (or interpolate, but drop is safer)
#   - outliers defined as > 3 standard deviations from mean -> flag and remove
#     (in practice you'd investigate before removing — mention this)
#   - returns outside [-50%, +50%] in a day are almost certainly data errors
#   - I'll return both the cleaned series and a log of what was removed

def clean_returns(returns, outlier_threshold=3, hard_limit=0.5):
    cleaned = []
    removed = []

    # first pass: remove Nones and hard-limit violations
    valid = []
    for i, r in enumerate(returns):
        if r is None:
            removed.append((i, r, 'missing'))
        elif abs(r) > hard_limit:
            removed.append((i, r, 'hard_limit'))
        else:
            valid.append((i, r))

    # second pass: remove statistical outliers from valid returns
    vals = [r for _, r in valid]
    if len(vals) > 1:
        mean = sum(vals) / len(vals)
        std  = math.sqrt(sum((r - mean)**2 for r in vals) / (len(vals) - 1))
        for i, r in valid:
            if abs(r - mean) > outlier_threshold * std:
                removed.append((i, r, f'outlier>{outlier_threshold}sd'))
            else:
                cleaned.append(r)
    else:
        cleaned = vals

    return cleaned, removed

messy = [0.01, -0.02, None, 0.03, 0.85, -0.01, 0.02, -0.15, None, 0.01, -0.003, 0.009]
clean, removed = clean_returns(messy)
print(f"Original length: {len(messy)}")
print(f"Cleaned length:  {len(clean)}")
print(f"Removed entries:")
for idx, val, reason in removed:
    print(f"  index {idx}: {val} -> {reason}")
