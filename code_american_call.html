American Call (Binomial Tree)

Source: american_call.py

import math

# ══════════════════════════════════════════════════════════════════════════════
# AMERICAN CALL PRICER — Binomial Tree (Cox-Ross-Rubinstein)
# ══════════════════════════════════════════════════════════════════════════════
# Black-Scholes only prices EUROPEAN options (fixed expiry, no early exercise).
# American options allow early exercise at any time -> need a tree/numerical method.
#
# Note: for a non-dividend paying stock, early exercise of a call is NEVER optimal
# (you lose time value). So American call = European call in that case.
# But the tree shows it correctly regardless — good to know and mention this.
#
# CRR parameters:
#   u = e^(sigma * sqrt(dt))        — up move factor
#   d = 1/u                         — down move factor (recombining tree)
#   p = (e^(r*dt) - d) / (u - d)   — risk-neutral probability of up move
#
# S = spot, K = strike, T = time to expiry (years), r = risk-free, sigma = vol
# steps = number of time steps in the tree

def american_call_binomial(S, K, T, r, sigma, steps=100):
    dt = T / steps
    u  = math.exp(sigma * math.sqrt(dt))    # up factor
    d  = 1 / u                              # down factor
    p  = (math.exp(r * dt) - d) / (u - d)  # risk-neutral up probability
    discount = math.exp(-r * dt)            # discount factor per step

    # build terminal stock prices at expiry (step = steps)
    # node (i) at terminal = S * u^i * d^(steps-i) for i = 0..steps
    stock = [S * (u ** i) * (d ** (steps - i)) for i in range(steps + 1)]

    # terminal option values = max(S - K, 0)
    option = [max(s - K, 0) for s in stock]

    # step backwards through the tree
    for step in range(steps - 1, -1, -1):
        for i in range(step + 1):
            s_node = S * (u ** i) * (d ** (step - i))
            hold   = discount * (p * option[i + 1] + (1 - p) * option[i])
            exercise = max(s_node - K, 0)           # intrinsic value
            option[i] = max(hold, exercise)         # american: take the best

    price = option[0]

    # delta: sensitivity of option price to <span class="nokatex">$</span>1 move in spot
    # computed from the first two nodes up and down from S
    s_up   = S * u
    s_down = S * d

    # one-step prices at up and down nodes
    # (re-use same tree logic for 1 step then discount back)
    v_up   = max(discount * (p * max(s_up * u - K, 0) + (1 - p) * max(s_up * d - K, 0)),
                 max(s_up - K, 0))
    v_down = max(discount * (p * max(s_down * u - K, 0) + (1 - p) * max(s_down * d - K, 0)),
                 max(s_down - K, 0))

    delta  = (v_up - v_down) / (s_up - s_down)
    # delta interpretation: if spot moves +<span class="nokatex">$</span>1, option value changes by delta dollars
    # call delta is between 0 and 1; ATM ~0.5; deep ITM -> 1; deep OTM -> 0

    return price, delta


S, K, T, r, sigma = 45, 50, 1, 0.05, 0.25

price, delta = american_call_binomial(S, K, T, r, sigma, steps=200)
print(f"American Call Price: ${price:.4f}")
print(f"Delta:               {delta:.4f}")
print()
print(f"Interpretation:")
print(f"  Spot = {S}, Strike = {K} -> OTM by ${K - S}")
print(f"  Delta = {delta:.4f} -> option gains ~${delta:.2f} per <span class="nokatex">$</span>1 rise in spot")
print(f"  To delta-hedge: short {delta:.4f} shares per long call")
print()

# verify: American call = European call for non-dividend stock
# (early exercise never optimal -> same price)
def european_call_bs(S, K, T, r, sigma):
    from math import log, exp, sqrt, erf
    d1 = (log(S/K) + (r + 0.5*sigma**2)*T) / (sigma*sqrt(T))
    d2 = d1 - sigma*sqrt(T)
    N  = lambda x: 0.5*(1 + erf(x/sqrt(2)))
    return S*N(d1) - K*exp(-r*T)*N(d2)

bs_price = european_call_bs(S, K, T, r, sigma)
print(f"European Call (BS):  ${bs_price:.4f}  <- should match American call (no dividends)")
