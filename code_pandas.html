import numpy as np
import pandas as pd
from scipy import stats

# ══════════════════════════════════════════════════════════════════════════════
# HISTORICAL VaR — pandas version
# ══════════════════════════════════════════════════════════════════════════════
# np.percentile does the sorting for you — cleaner and handles edge cases

def historical_var_np(returns, confidence=0.95):
    # percentile at (1-confidence)*100 = the loss threshold
    return np.percentile(returns, (1 - confidence) * 100)

returns = np.array([0.01, -0.02, 0.03, -0.05, 0.02, -0.01, 0.04, -0.03, 0.00, -0.04])
print(f"95% Historical VaR (numpy): {historical_var_np(returns):.2%}")


# ══════════════════════════════════════════════════════════════════════════════
# PARAMETRIC VaR — scipy version (no hardcoded z-scores needed)
# ══════════════════════════════════════════════════════════════════════════════
# scipy.stats.norm.ppf gives the inverse CDF (percent point function)
# ppf(0.05) = -1.645 automatically — no lookup table needed

def parametric_var_scipy(returns, confidence=0.95):
    mean = np.mean(returns)
    std  = np.std(returns, ddof=1)                  # ddof=1 = sample std (n-1)
    z = stats.norm.ppf(1 - confidence)              # e.g. ppf(0.05) = -1.645
    return mean + z * std                           # negative = loss

print(f"95% Parametric VaR (scipy): {parametric_var_scipy(returns):.2%}")
print(f"99% Parametric VaR (scipy): {parametric_var_scipy(returns, 0.99):.2%}")


# ══════════════════════════════════════════════════════════════════════════════
# MAX DRAWDOWN — pandas version
# ══════════════════════════════════════════════════════════════════════════════

def max_drawdown_pd(prices):
    prices = pd.Series(prices)
    rolling_peak = prices.cummax()                  # running peak at each point
    drawdown = (prices - rolling_peak) / rolling_peak
    return drawdown.min()                           # most negative value

prices = [100, 110, 105, 120, 90, 95, 80, 115, 145, 60, 100]
print(f"Max Drawdown (pandas): {max_drawdown_pd(prices):.2%}")


# ══════════════════════════════════════════════════════════════════════════════
# ROLLING VOLATILITY — pandas version
# ══════════════════════════════════════════════════════════════════════════════

def rolling_vol_pd(prices, window=20):
    prices  = pd.Series(prices)
    returns = prices.pct_change().dropna()          # daily returns
    rolling_std = returns.rolling(window).std()     # sample std by default (ddof=1)
    annualized  = rolling_std * np.sqrt(252)
    return annualized

# generate fake price series for demo
np.random.seed(42)
fake_prices = 100 * np.cumprod(1 + np.random.normal(0.0005, 0.01, 60))
vol = rolling_vol_pd(fake_prices, window=10)
print(f"\nRolling Vol (last 5 values):\n{vol.dropna().tail()}")


# ══════════════════════════════════════════════════════════════════════════════
# ROLLING SHARPE — pandas version
# ══════════════════════════════════════════════════════════════════════════════

def rolling_sharpe_pd(prices, window=20, risk_free=0.0):
    prices  = pd.Series(prices)
    returns = prices.pct_change().dropna()
    excess  = returns - risk_free / 252             # daily excess return
    roll_mean = excess.rolling(window).mean()
    roll_std  = excess.rolling(window).std()        # ddof=1 by default
    sharpe    = (roll_mean / roll_std) * np.sqrt(252)
    return sharpe

sharpe = rolling_sharpe_pd(fake_prices, window=10)
print(f"\nRolling Sharpe (last 5 values):\n{sharpe.dropna().tail()}")

# BREACH DETECTION — pandas version
# ══════════════════════════════════════════════════════════════════════════════

def detect_breaches_pd(pnl, limit):
    pnl = pd.Series(pnl)
    breaches = pnl[pnl < limit]
    return breaches                                 # Series with original indices

def rolling_breach_pd(pnl, limit, window=3):
    pnl = pd.Series(pnl)
    cumulative = pnl.rolling(window).sum()
    return cumulative[cumulative < limit]

pnl = [10000, -20000, 5000, -60000, 3000, -55000, 8000, -51000, -66000, 156000]
print(f"\nBreaches:\n{detect_breaches_pd(pnl, -50000)}")
print(f"\nRolling 3-day breaches:\n{rolling_breach_pd(pnl, -50000, window=3)}")

# OHLC ANALYSIS — if given a price DataFrame
# ══════════════════════════════════════════════════════════════════════════════
# Most likely format: columns = [Open, High, Low, Close, Volume]

ohlc = pd.DataFrame({
    'Open':   [100, 102, 101, 105, 103],
    'High':   [104, 106, 103, 108, 107],
    'Low':    [99,  101, 99,  104, 101],
    'Close':  [102, 101, 105, 103, 106],
    'Volume': [1e6, 1.2e6, 0.9e6, 1.5e6, 1.1e6]
})

ohlc['return']       = ohlc['Close'].pct_change()
ohlc['intraday_rng'] = (ohlc['High'] - ohlc['Low']) / ohlc['Open']  # intraday range %
ohlc['true_range']   = np.maximum(
                            ohlc['High'] - ohlc['Low'],
                            np.maximum(
                                abs(ohlc['High'] - ohlc['Close'].shift(1)),
                                abs(ohlc['Low']  - ohlc['Close'].shift(1))
                            )
                        )
ohlc['vwap']         = (ohlc['Close'] * ohlc['Volume']).cumsum() / ohlc['Volume'].cumsum()
ohlc['drawdown']     = (ohlc['Close'] / ohlc['Close'].cummax()) - 1

print(f"\nOHLC with risk metrics:\n{ohlc.to_string()}")

# CORRELATION MATRIX — numpy version
# ══════════════════════════════════════════════════════════════════════════════
# Risk managers use correlation to assess portfolio diversification
# Correlation breakdown in crises: assets that were uncorrelated suddenly move together

np.random.seed(0)
asset_returns = pd.DataFrame({
    'SPX':   np.random.normal(0.0003, 0.01, 100),
    'Bonds': np.random.normal(0.0001, 0.005, 100),
    'Gold':  np.random.normal(0.0002, 0.008, 100),
})

corr_matrix = asset_returns.corr()
print(f"\nCorrelation Matrix:\n{corr_matrix.round(3)}")


# ══════════════════════════════════════════════════════════════════════════════
# PORTFOLIO VaR — with weights and covariance matrix
# ══════════════════════════════════════════════════════════════════════════════
# Parametric portfolio VaR: sigma_p = sqrt(w^T * Cov * w)
# Then VaR = portfolio_value * z * sigma_p

def portfolio_var(weights, returns_df, confidence=0.95, portfolio_value=1e6):
    weights = np.array(weights)
    cov     = returns_df.cov().values * 252         # annualized covariance
    sigma_p = np.sqrt(weights @ cov @ weights)      # portfolio vol (annualized)
    z       = stats.norm.ppf(1 - confidence)
    daily_sigma = sigma_p / np.sqrt(252)            # back to daily
    return portfolio_value * z * daily_sigma        # negative = loss in $

weights = [0.5, 0.3, 0.2]
var_dollar = portfolio_var(weights, asset_returns)
print(f"\n1-day 95% Portfolio VaR: ${var_dollar:,.0f}")
