def historical_var(returns, confidence=0.95):
    sorted_returns = sorted(returns)           # sort ascending: worst losses first
    index = int((1 - confidence) * len(sorted_returns))  # 5% tail cutoff index
    # e.g. 100 returns, 95% confidence -> index=5 -> 5 days were worse, 95 were better
    # limitation: needs 250+ observations; with small samples index=0 (absolute worst day)
    return sorted_returns[index]

returns = [0.01, -0.02, 0.03, -0.05, 0.02, -0.01, 0.04, -0.03, 0.00, -0.04]
var_95 = historical_var(returns, confidence=0.95)
print(f"95% Historical VaR: {var_95:.2%}")

def max_drawdown(prices):
    peak = prices[0]
    max_dd = 0.0
    for price in prices:
        if price > peak:
            peak = price                        # update running peak
        drawdown = (price - peak) / peak       # pct decline from peak (always <= 0)
        if drawdown < max_dd:                  # track worst (most negative) drawdown
            max_dd = drawdown
    return max_dd
    # if given returns not prices: prices = [100]; [prices.append(prices[-1]*(1+r)) for r in returns]


prices = [100, 110, 105, 120, 90, 95, 80, 115, 145, 60, 100, 0]
print(f"Max Drawdown: {max_drawdown(prices):.2%}")

def detect_breaches(pnl, limit):
    # day-by-day breach: single day loss exceeds limit
    breaches = []
    for i, value in enumerate(pnl):
        if value < limit:
            breaches.append((i, value))
    return breaches

def rolling_breach(pnl, limit, window=3):
    # cumulative breach: sum of last `window` days exceeds limit
    # window=1 is equivalent to detect_breaches; window=0 is invalid
    breaches = []
    for i in range(window - 1, len(pnl)):
        cumulative = sum(pnl[i - window + 1 : i + 1])
        if cumulative < limit:
            breaches.append((i, cumulative))
    return breaches

pnl = [10000, -20000, 5000, -60000, 3000, -55000, 8000, -51000, -66000, 156000]
limit = -50000

breaches = detect_breaches(pnl, limit)
breaches = rolling_breach(pnl, limit, window=1)
for day, loss in breaches:
    print(f"Day {day}: P&L = {loss:,}  BREACH")

import math
def rolling_sharpe(returns, window, risk_free=0.0):
    result = []
    for i in range(window - 1, len(returns)):
        w = returns[i - window + 1 : i + 1]
        mean = sum(w) / window
        excess = [r - risk_free / 252 for r in w]  # daily risk-free rate = annual / 252
        variance = sum((r - mean) ** 2 for r in excess) / (window - 1)  # n-1: sample std
        # n-1 (Bessel's correction): window is a sample, not full population
        # mean already estimated from same data -> dividing by n underestimates true variance
        std = math.sqrt(variance)
        if std == 0:                               # flat returns: Sharpe undefined
            result.append(None)
        else:
            result.append((mean / std) * math.sqrt(252))  # annualize by sqrt(252)
    return result

returns = [0.01, -0.005, 0.02, -0.01, 0.015, 0.005, -0.008, 0.012]
print(rolling_sharpe(returns, window=4))


# ── PARAMETRIC VaR ────────────────────────────────────────────────────────────
# Assumes returns are normally distributed. VaR = mean - z * std
# z-score: 1.645 for 95%, 2.326 for 99%
# Faster than historical but wrong when returns are fat-tailed (which they usually are)
def normal_cdf(x):
    # math.erf is the error function; this gives Phi(x) without scipy
    return 0.5 * (1 + math.erf(x / math.sqrt(2)))

def parametric_var(returns, confidence=0.95):
    n = len(returns)
    mean = sum(returns) / n
    std = math.sqrt(sum((r - mean) ** 2 for r in returns) / (n - 1))  # sample std
    # find z such that normal_cdf(z) = confidence -> binary search or hardcode
    # hardcoded z-scores for common confidence levels:
    z_scores = {0.90: 1.282, 0.95: 1.645, 0.99: 2.326}
    z = z_scores.get(confidence, 1.645)
    return mean - z * std  # negative number = loss

returns = [0.01, -0.02, 0.03, -0.05, 0.02, -0.01, 0.04, -0.03, 0.00, -0.04]
print(f"95% Parametric VaR: {parametric_var(returns, 0.95):.2%}")
print(f"99% Parametric VaR: {parametric_var(returns, 0.99):.2%}")


# ── PORTFOLIO P&L ─────────────────────────────────────────────────────────────
# Given weights and individual asset returns, compute weighted portfolio return
# weights must sum to 1 (long-only) or can be +/- (long-short)
def portfolio_pnl(weights, returns):
    # dot product: sum of weight_i * return_i for each asset
    return sum(w * r for w, r in zip(weights, returns))

weights = [0.4, 0.3, 0.3]          # 40% asset A, 30% B, 30% C
asset_returns = [0.02, -0.01, 0.03]
print(f"Portfolio Return: {portfolio_pnl(weights, asset_returns):.2%}")


# ── CORRELATION ───────────────────────────────────────────────────────────────
# Pearson correlation between two return series, from scratch (no numpy)
# corr = cov(X,Y) / (std(X) * std(Y))
def correlation(x, y):
    n = len(x)
    mean_x = sum(x) / n
    mean_y = sum(y) / n
    cov = sum((x[i] - mean_x) * (y[i] - mean_y) for i in range(n)) / (n - 1)
    std_x = math.sqrt(sum((v - mean_x) ** 2 for v in x) / (n - 1))
    std_y = math.sqrt(sum((v - mean_y) ** 2 for v in y) / (n - 1))
    if std_x == 0 or std_y == 0:
        return None  # undefined if one series is flat
    return cov / (std_x * std_y)

x = [0.01, -0.02, 0.03, -0.01, 0.02]
y = [0.02, -0.01, 0.025, -0.015, 0.01]
print(f"Correlation: {correlation(x, y):.4f}")  # expect close to 1 (similar series)


# ── MONTE CARLO VaR ───────────────────────────────────────────────────────────
# Simulate N random return paths, take percentile — no normality assumption needed
# Uses Box-Muller to generate normal random numbers without numpy
import random

def monte_carlo_var(mean, std, n_simulations=100000, confidence=0.95):
    simulated = []
    for _ in range(n_simulations):
        # Box-Muller transform: convert two uniform randoms to one normal random
        u1 = random.random()
        u2 = random.random()
        z = math.sqrt(-2 * math.log(u1)) * math.cos(2 * math.pi * u2)
        simulated.append(mean + std * z)
    simulated.sort()
    index = int((1 - confidence) * n_simulations)
    return simulated[index]

# use same mean/std as the parametric example
returns = [0.01, -0.02, 0.03, -0.05, 0.02, -0.01, 0.04, -0.03, 0.00, -0.04]
n = len(returns)
mean = sum(returns) / n
std = math.sqrt(sum((r - mean) ** 2 for r in returns) / (n - 1))
print(f"95% Monte Carlo VaR: {monte_carlo_var(mean, std):.2%}")
# result will be close to parametric VaR since we're sampling from same normal distribution
