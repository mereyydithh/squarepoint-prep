class RiskLimit:
    """Risk limit with automatic breach detection."""

    def __init__(self, limit_name: str, max_value: float):
        self._name = limit_name
        self._max_value = max_value
        self._current_value = 0.0
        self._breach_history: list[dict] = []

    @property
    def utilization(self) -> float:
        """Limit utilization as percentage."""
        return (self._current_value / self._max_value) * 100 if self._max_value else 0

    def update(self, new_value: float) -> bool:
        """Update current value and return True if limit is breached."""
        self._current_value = new_value
        breached = self._current_value > self._max_value
        if breached:
            self._breach_history.append({
                "timestamp": datetime.now(),
                "value": new_value,
                "limit": self._max_value,
                "excess": new_value - self._max_value
            })
        return breached

    @property
    def rag_status(self) -> str:
        """Return RAG status based on utilization."""
        util = self.utilization
        if util < 70:
            return "GREEN"
        elif util < 90:
            return "AMBER"
        return "RED"

Why encapsulation matters for RM: The breach history is private — no one can accidentally modify it. The rag_status is computed (not stored), so it's always consistent with the current value.

2. Inheritance — Parent → Child

from abc import ABC, abstractmethod

class Instrument(ABC):
    """Base class for all financial instruments."""

    def __init__(self, ticker: str, quantity: float, price: float):
        self.ticker = ticker
        self.quantity = quantity
        self.price = price

    @abstractmethod
    def market_value(self) -> float:
        """Current market value in fund currency."""
        pass

    @abstractmethod
    def delta_exposure(self) -> float:
        """First-order sensitivity to underlying price."""
        pass

class Equity(Instrument):
    def market_value(self) -> float:
        return self.quantity * self.price

    def delta_exposure(self) -> float:
        return self.market_value()  # Delta-1 for equities

class Bond(Instrument):
    def __init__(self, ticker: str, face_value: float, clean_price: float,
                 accrued: float, modified_duration: float):
        super().__init__(ticker, face_value, clean_price)
        self.accrued = accrued
        self.mod_duration = modified_duration

    def market_value(self) -> float:
        return self.quantity * (self.price + self.accrued) / 100

    def delta_exposure(self) -> float:
        return -self.market_value() * self.mod_duration  # DV01 direction

    def dv01(self) -> float:
        """Dollar value of 1bp rate move."""
        return self.market_value() * self.mod_duration * 0.0001

class Option(Instrument):
    def __init__(self, ticker: str, quantity: float, price: float,
                 delta: float, gamma: float, vega: float, theta: float,
                 underlying_price: float, multiplier: int = 100):
        super().__init__(ticker, quantity, price)
        self.delta = delta
        self.gamma = gamma
        self.vega = vega
        self.theta = theta
        self.underlying_price = underlying_price
        self.multiplier = multiplier

    def market_value(self) -> float:
        return self.quantity * self.price * self.multiplier

    def delta_exposure(self) -> float:
        return self.quantity * self.delta * self.underlying_price * self.multiplier

    def gamma_pnl(self, price_move_pct: float) -> float:
        """P&L from gamma for a given % price move."""
        dollar_move = self.underlying_price * price_move_pct
        return 0.5 * self.quantity * self.gamma * dollar_move**2 * self.multiplier

VaR — Three Methods

import numpy as np
import pandas as pd
from scipy import stats

def parametric_var(returns: pd.Series, confidence: float = 0.99,
                   portfolio_value: float = 1e6) -> float:
    """Parametric (variance-covariance) VaR assuming normal distribution."""
    mu = returns.mean()
    sigma = returns.std()
    z = stats.norm.ppf(1 - confidence)  # z = -2.326 for 99%
    var_return = mu + z * sigma
    return -var_return * portfolio_value

def historical_var(returns: pd.Series, confidence: float = 0.99,
                   portfolio_value: float = 1e6) -> float:
    """Historical simulation VaR — no distribution assumption."""
    cutoff = returns.quantile(1 - confidence)
    return -cutoff * portfolio_value

def monte_carlo_var(mu: float, sigma: float, portfolio_value: float = 1e6,
                    confidence: float = 0.99, n_sims: int = 100_000) -> float:
    """Monte Carlo VaR with normal returns."""
    simulated = np.random.normal(mu, sigma, n_sims)
    simulated_pnl = portfolio_value * simulated
    return -np.percentile(simulated_pnl, (1 - confidence) * 100)

Expected Shortfall (CVaR)

def expected_shortfall(returns: pd.Series, confidence: float = 0.99,
                       portfolio_value: float = 1e6) -> float:
    """ES = average loss beyond VaR. Better tail risk measure than VaR."""
    cutoff = returns.quantile(1 - confidence)
    tail_returns = returns[returns <= cutoff]
    return -tail_returns.mean() * portfolio_value

# Compare VaR vs ES
var_99 = historical_var(returns, 0.99)
es_99 = expected_shortfall(returns, 0.99)
tail_ratio = es_99 / var_99  # >1.5 → fat tail warning

Stress Testing Engine

def stress_test(positions: pd.DataFrame, scenarios: dict[str, dict]) -> pd.DataFrame:
    """
    Apply stress scenarios to a portfolio.

    scenarios = {
        "Rates +100bp": {"rates_shock": 0.01, "equity_shock": -0.05, "vol_shock": 0.3},
        "COVID Redux": {"rates_shock": -0.015, "equity_shock": -0.30, "vol_shock": 1.5},
        "Oil Spike": {"oil_shock": 0.40, "equity_shock": -0.08, "vol_shock": 0.2},
    }
    """
    results = {}
    for scenario_name, shocks in scenarios.items():
        pnl = 0.0
        for _, pos in positions.iterrows():
            # Delta P&L from equity shock
            if "equity_shock" in shocks and pos["asset_class"] == "equity":
                pnl += pos["market_value"] * pos["beta"] * shocks["equity_shock"]
            # DV01 P&L from rates shock
            if "rates_shock" in shocks and pos["asset_class"] == "bond":
                pnl += pos["dv01"] * shocks["rates_shock"] * 10_000  # DV01 per bp
            # Vega P&L from vol shock
            if "vol_shock" in shocks and pos.get("vega", 0) != 0:
                pnl += pos["vega"] * shocks["vol_shock"]
            # Gamma P&L (second order)
            if "equity_shock" in shocks and pos.get("gamma", 0) != 0:
                dollar_move = pos["underlying_price"] * shocks["equity_shock"]
                pnl += 0.5 * pos["gamma"] * dollar_move**2
        results[scenario_name] = pnl

    return pd.DataFrame.from_dict(results, orient="index", columns=["Stress P&L"])

Drawdown Analysis

def drawdown_analysis(cumulative_pnl: pd.Series) -> pd.DataFrame:
    """Identify all drawdown periods with peak, trough, recovery, and duration."""
    peak = cumulative_pnl.cummax()
    drawdown = cumulative_pnl - peak
    drawdown_pct = drawdown / peak.replace(0, np.nan)

    # Identify drawdown periods
    in_drawdown = drawdown < 0
    dd_groups = (~in_drawdown).cumsum()

    results = []
    for group_id, group in cumulative_pnl[in_drawdown].groupby(dd_groups):
        if len(group) == 0:
            continue
        start = group.index[0]
        trough_date = group.idxmin()
        trough_value = drawdown.loc[trough_date]
        peak_value = peak.loc[start]
        dd_pct = trough_value / peak_value * 100

        # Find recovery date
        post_trough = cumulative_pnl.loc[trough_date:]
        recovered = post_trough[post_trough >= peak_value]
        recovery_date = recovered.index[0] if len(recovered) > 0 else None
        duration = (group.index[-1] - start).days

        results.append({
            "start": start,
            "trough_date": trough_date,
            "max_drawdown": trough_value,
            "drawdown_pct": dd_pct,
            "recovery_date": recovery_date,
            "duration_days": duration,
        })

    return pd.DataFrame(results).sort_values("max_drawdown").head(10)

def pnl_attribution(positions: pd.DataFrame, 
                     market_moves: dict[str, float]) -> pd.DataFrame:
    """
    Decompose daily P&L into risk factor contributions.

    market_moves = {"SPX": -0.012, "rates_10y": 0.0005, "vol": 0.02, "oil": -0.03}
    """
    positions = positions.copy()

    # Delta P&L
    positions["delta_pnl"] = positions["delta_exposure"] * market_moves.get("SPX", 0)

    # Gamma P&L
    spx_move_dollar = positions["underlying_price"] * market_moves.get("SPX", 0)
    positions["gamma_pnl"] = 0.5 * positions["gamma_exposure"] * spx_move_dollar**2

    # Rates P&L (DV01 × bp move)
    bp_move = market_moves.get("rates_10y", 0) * 10_000
    positions["rates_pnl"] = -positions["dv01"] * bp_move

    # Vega P&L
    positions["vega_pnl"] = positions["vega_exposure"] * market_moves.get("vol", 0)

    # Theta P&L (daily)
    positions["theta_pnl"] = positions["theta_exposure"]

    # Unexplained (residual)
    positions["total_pnl"] = positions["actual_pnl"]
    explained = positions<em>"delta_pnl", "gamma_pnl", "rates_pnl", "vega_pnl", "theta_pnl"</em>.sum(axis=1)
    positions["unexplained"] = positions["total_pnl"] - explained

    return positions

Correlation and Factor Analysis

from sklearn.decomposition import PCA
from sklearn.linear_model import LinearRegression

def rolling_correlation(returns_a: pd.Series, returns_b: pd.Series,
                        window: int = 60) -> pd.Series:
    """Rolling correlation between two return series."""
    return returns_a.rolling(window).corr(returns_b)

def factor_decomposition(pm_returns: pd.Series, 
                          factor_returns: pd.DataFrame) -> dict:
    """
    Regress PM returns on factor returns.
    Returns alpha, betas, R², and factor attribution.
    """
    # Align dates
    aligned = pd.concat([pm_returns, factor_returns], axis=1).dropna()
    y = aligned.iloc[:, 0].values
    X = aligned.iloc[:, 1:].values

    model = LinearRegression().fit(X, y)

    r_squared = model.score(X, y)
    residuals = y - model.predict(X)

    # Annualize alpha
    daily_alpha = model.intercept_
    annual_alpha = daily_alpha * 252

    # T-statistic for alpha
    se_alpha = residuals.std() / np.sqrt(len(y))
    t_stat = daily_alpha / se_alpha

    return {
        "alpha_daily_bps": daily_alpha * 10_000,
        "alpha_annual_pct": annual_alpha * 100,
        "t_stat_alpha": t_stat,
        "betas": dict(zip(factor_returns.columns, model.coef_)),
        "r_squared": r_squared,
        "idiosyncratic_ratio": 1 - r_squared,
    }

def pca_risk_decomposition(returns: pd.DataFrame, n_components: int = 5) -> dict:
    """PCA decomposition of portfolio risk factors."""
    pca = PCA(n_components=n_components)
    pca.fit(returns.dropna())

    return {
        "explained_variance_ratio": pca.explained_variance_ratio_,
        "cumulative_explained": np.cumsum(pca.explained_variance_ratio_),
        "loadings": pd.DataFrame(
            pca.components_.T,
            index=returns.columns,
            columns=[f"PC{i+1}" for i in range(n_components)]
        ),
    }

Monte Carlo — Multi-Asset with Correlation

def monte_carlo_portfolio_var(returns: pd.DataFrame, weights: np.ndarray,
                               portfolio_value: float = 1e7,
                               confidence: float = 0.99,
                               n_sims: int = 50_000,
                               use_t_dist: bool = False,
                               t_df: int = 5) -> dict:
    """
    Multi-asset Monte Carlo VaR with optional fat tails (Student-t).
    Uses Cholesky decomposition for correlated simulation.
    """
    n_assets = returns.shape[1]
    mu = returns.mean().values
    cov = returns.cov().values
    L = np.linalg.cholesky(cov)

    # Generate random scenarios
    if use_t_dist:
        z = stats.t.rvs(df=t_df, size=(n_sims, n_assets))
    else:
        z = np.random.normal(size=(n_sims, n_assets))

    # Apply correlation structure
    correlated_returns = mu + z @ L.T

    # Portfolio-level P&L
    portfolio_returns = correlated_returns @ weights
    portfolio_pnl = portfolio_value * portfolio_returns

    var = -np.percentile(portfolio_pnl, (1 - confidence) * 100)
    es = -portfolio_pnl[portfolio_pnl <= -var].mean() if any(portfolio_pnl <= -var) else var

    return {
        "VaR_99": var,
        "ES_99": es,
        "ES_VaR_ratio": es / var if var > 0 else 0,
        "mean_pnl": portfolio_pnl.mean(),
        "worst_case": -portfolio_pnl.min(),
        "simulation_count": n_sims,
        "fat_tails": use_t_dist,
    }

Sharpe, Sortino, and Performance Metrics

def performance_metrics(returns: pd.Series, risk_free_rate: float = 0.04) -> dict:
    """Comprehensive performance metrics suite."""
    daily_rf = risk_free_rate / 252
    excess = returns - daily_rf

    # Sharpe
    sharpe = excess.mean() / excess.std() * np.sqrt(252)

    # Sortino (downside deviation only)
    downside = excess[excess < 0]
    downside_std = np.sqrt((downside**2).mean())
    sortino = excess.mean() / downside_std * np.sqrt(252) if downside_std > 0 else 0

    # Max drawdown
    cumulative = (1 + returns).cumprod()
    peak = cumulative.cummax()
    max_dd = ((cumulative - peak) / peak).min()

    # Calmar ratio (return / max drawdown)
    annual_return = returns.mean() * 252
    calmar = annual_return / abs(max_dd) if max_dd != 0 else 0

    # Skewness and kurtosis
    skew = returns.skew()
    kurt = returns.kurtosis()  # Excess kurtosis

    # Hit rate
    hit_rate = (returns > 0).mean()

    # Best/worst days
    best = returns.max()
    worst = returns.min()

    return {
        "annual_return_pct": annual_return * 100,
        "annual_vol_pct": returns.std() * np.sqrt(252) * 100,
        "sharpe": sharpe,
        "sortino": sortino,
        "max_drawdown_pct": max_dd * 100,
        "calmar": calmar,
        "skewness": skew,
        "excess_kurtosis": kurt,
        "hit_rate_pct": hit_rate * 100,
        "best_day_pct": best * 100,
        "worst_day_pct": worst * 100,
        "trading_days": len(returns),
    }

-- PM performance ranking with Sharpe-like metric
WITH pm_stats AS (
    SELECT pm_name,
           AVG(daily_pnl) AS avg_daily_pnl,
           STDDEV(daily_pnl) AS daily_vol,
           COUNT(*) AS trading_days,
           SUM(daily_pnl) AS total_pnl,
           MIN(daily_pnl) AS worst_day,
           SUM(CASE WHEN daily_pnl > 0 THEN 1 ELSE 0 END)::FLOAT / COUNT(*) AS hit_rate
    FROM daily_pnl
    WHERE date >= DATE '2026-01-01'
    GROUP BY pm_name
),

    SELECT *,
           avg_daily_pnl / NULLIF(daily_vol, 0) * SQRT(252) AS sharpe_approx,
           RANK() OVER (ORDER BY avg_daily_pnl / NULLIF(daily_vol, 0) DESC) AS sharpe_rank,
           RANK() OVER (ORDER BY total_pnl DESC) AS pnl_rank
    FROM pm_stats
)

       ROUND(total_pnl / 1e6, 2) AS total_pnl_mm,
       ROUND(sharpe_approx, 2) AS sharpe,
       ROUND(hit_rate * 100, 1) AS hit_rate_pct,
       ROUND(worst_day / 1e6, 2) AS worst_day_mm,
       sharpe_rank,
       pnl_rank
FROM pm_rankings
ORDER BY sharpe_rank;

WITH pm_positions AS (
    SELECT pm_name,
           ticker,
           ABS(quantity * last_price) AS abs_mv,

  • Column-oriented: Aggregations over billions of rows in milliseconds (vs. minutes in SQL)

        return results

import functools
import time
import logging

def timer(func):
    """Measure execution time of any function."""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        logging.info(f"{func.__name__} took {elapsed:.3f}s")
        return result
    return wrapper

def validate_returns(func):
    """Validate that return series is clean before computation."""
    @functools.wraps(func)
    def wrapper(returns: pd.Series, *args, **kwargs):
        if returns.isna().any():
            returns = returns.dropna()
            logging.warning(f"Dropped {returns.isna().sum()} NaN values")
        if len(returns) < 30:
            raise ValueError(f"Insufficient data: {len(returns)} < 30 observations")
        return func(returns, *args, **kwargs)
    return wrapper

@timer
@validate_returns
def compute_var(returns: pd.Series, confidence: float = 0.99) -> float:
    return -returns.quantile(1 - confidence)

def rolling_var(returns: pd.Series, window: int = 252, 
                confidence: float = 0.99):
    """Memory-efficient rolling VaR using generator."""
    for i in range(window, len(returns)):
        window_returns = returns.iloc[i - window:i]
        yield returns.index[i], -window_returns.quantile(1 - confidence)

def find_pairs_summing_to(arr: list[float], target: float) -> list[tuple]:
    """Find all pairs in sorted array that sum to target."""
    arr.sort()
    left, right = 0, len(arr) - 1
    pairs = []
    while left < right:
        total = arr[left] + arr[right]
        if total == target:
            pairs.append((arr[left], arr[right]))
            left += 1
            right -= 1
        elif total < target:
            left += 1
        else:
            right -= 1
    return pairs

Pattern 2: Sliding Window

def max_drawdown_window(pnl: list[float], window: int) -> float:
    """Maximum drawdown within any rolling window."""
    max_dd = 0
    for i in range(len(pnl) - window + 1):
        w = pnl[i:i + window]
        peak = w[0]
        dd = 0
        for val in w:
            peak = max(peak, val)
            dd = min(dd, val - peak)
        max_dd = min(max_dd, dd)
    return max_dd

    """HHI concentration index by sector."""
    sector_mv = Counter()
    total_mv = 0
    for p in positions:
        mv = abs(p["market_value"])
        sector_mv[p["sector"]] += mv
        total_mv += mv

    hhi = sum((mv / total_mv) ** 2 for mv in sector_mv.values())

import bisect

def find_var_breach_date(sorted_pnl_dates: list[tuple], var_limit: float) -> int:
    """Find first date where cumulative P&L breaches VaR limit.
    sorted_pnl_dates: [(date, cumulative_pnl), ...] sorted by date.
    """

Pattern 5: Dynamic Programming

def max_profit_k_trades(prices: list[float], k: int) -> float:
    """Maximum profit with at most k buy-sell transactions."""
    if not prices or k == 0:
        return 0

    n = len(prices)
    if k >= n // 2:
        # Unlimited trades
        return sum(max(prices[i+1] - prices[i], 0) for i in range(n-1))

    # DP: dp[j][i] = max profit using at most j transactions up to day i
    dp = [[0] * n for _ in range(k + 1)]
    for j in range(1, k + 1):
        best_buy = -prices[0]
        for i in range(1, n):
            dp[j][i] = max(dp[j][i-1], prices[i] + best_buy)
            best_buy = max(best_buy, dp[j-1][i] - prices[i])

    return dp[k][n-1]

from functools import lru_cache

@lru_cache(maxsize=None)
def expected_flips_consecutive_heads(n: int) -> float:
    """Expected number of coin flips to get n consecutive heads."""
    if n == 0:
        return 0
    if n == 1:
        return 2  # Geometric: E[X] = 1/p = 1/0.5 = 2
    # After getting n-1 consecutive heads (expected E(n-1) flips):
    # - Heads (p=0.5): done! Total = E(n-1) + 1
    # - Tails (p=0.5): restart entirely. Total = E(n-1) + 1 + E(n)
    # E(n) = 0.5 * (E(n-1) + 1) + 0.5 * (E(n-1) + 1 + E(n))
    # E(n) = E(n-1) + 1 + 0.5 * E(n)
    # 0.5 * E(n) = E(n-1) + 1
    # E(n) = 2 * (E(n-1) + 1)
    return 2 * (expected_flips_consecutive_heads(n - 1) + 1)

# E(1) = 2, E(2) = 6, E(3) = 14, E(4) = 30, E(5) = 62

import logging
from typing import Optional

logger = logging.getLogger("risk_engine")

def safe_var_calculation(returns: pd.Series, 
                          confidence: float = 0.99,
                          portfolio_value: float = 1e6,
                          min_observations: int = 100) -> Optional[float]:
    """VaR with comprehensive input validation and error handling."""

    # Input validation
    if returns is None or len(returns) == 0:
        logger.error("Empty return series provided to VaR calculation")
        return None

    if not 0.9 <= confidence <= 0.999:
        raise ValueError(f"Confidence {confidence} outside valid range [0.9, 0.999]")

    if portfolio_value <= 0:
        raise ValueError(f"Portfolio value must be positive, got {portfolio_value}")

    # Data quality checks
    nan_count = returns.isna().sum()
    if nan_count > 0:
        logger.warning(f"Dropping {nan_count} NaN values from {len(returns)} observations")
        returns = returns.dropna()

    if len(returns) < min_observations:
        logger.warning(f"Only {len(returns)} observations (min: {min_observations})")
        return None

    # Outlier check (not removal — log and flag)
    z_scores = (returns - returns.mean()) / returns.std()
    extreme_count = (z_scores.abs() > 5).sum()
    if extreme_count > 0:
        logger.warning(f"{extreme_count} observations exceed 5-sigma — check data quality")

    # Calculate
    var_return = returns.quantile(1 - confidence)
    var_dollar = -var_return * portfolio_value

    # Sanity check
    if var_dollar < 0:
        logger.error(f"Negative VaR ({var_dollar}) — this shouldn't happen. Check return sign convention.")
        return None

    if var_dollar > portfolio_value * 0.5:
        logger.warning(f"VaR ({var_dollar:,.0f}) exceeds 50% of portfolio value — verify inputs")

    return var_dollar

Q: "How would you build a daily risk report?"

"I'd build a Python pipeline with four stages: ingestion, calculation, limits, and reporting.

Ingestion: SQL queries to pull positions from the risk database, joined with market data and reference data. I'd use pandas for the joins and cleaning. Reconciliation against PB statements as a data quality gate.

Calculation: VaR using historical simulation — sort the last 500 days of returns, take the 1st percentile. Stress tests using pre-defined scenarios — rates ±100bp, equity ±20%, vol ×2.

Limits: Compare each metric against its limit. RAG status: green under 70% utilization, amber 70-90%, red above 90%. Any red → automatic escalation email.

Reporting: Summary dashboard for the CRO with total VaR, P&L attribution by factor, limit status. Detailed PM-level views. All metrics computed by 7:30 AM for the morning risk meeting.

The architecture uses the Strategy pattern — each risk model implements a common interface so new calculations can be added without modifying the pipeline."

Q: "Write a function to compute VaR and explain your design choices."

"Here's how I'd implement historical VaR:

I'd take a return series and confidence level as inputs, validate that I have enough data — at least 100 observations — and drop NaN values with a warning. Then I'd compute the quantile at the 1-minus-confidence level and multiply by portfolio value.

Design choices: I use historical simulation rather than parametric because it doesn't assume normality — important since real returns have fat tails. I'd also compute Expected Shortfall alongside VaR — it's the average loss beyond VaR, which captures the tail shape that VaR misses. If the ES-to-VaR ratio exceeds 1.5, I'd flag that the tail is unusually fat.

For a production system, I'd add input validation, NaN handling, sanity checks — VaR shouldn't be negative, shouldn't exceed 50% of portfolio value — and structured logging. And I'd write unit tests: known distributions where VaR has an analytical solution, monotonicity checks — higher vol should give higher VaR — and edge cases like empty series."

Q: "What SQL would you write to find which PMs have the most concentrated books?"

"I'd use a CTE to compute each PM's position-level market values and total book size, then calculate the HHI — Herfindahl-Hirschman Index — which is the sum of squared position weights. A perfectly diversified book of 100 equal positions has HHI of 0.01. A single-name book has HHI of 1.0.

I'd flag any PM with HHI above 0.15 — that implies an effective position count of about 6 or fewer. I'd also show the max single position as a percentage of their book. Any PM where one name is more than 20% of the book deserves a conversation about concentration risk.

The query would join positions to market data for current prices, group by PM and ticker, compute absolute market values, then aggregate into HHI at the PM level. Sorted descending by HHI so the most concentrated books are at the top."
