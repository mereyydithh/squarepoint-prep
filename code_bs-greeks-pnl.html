Black-Scholes, Greeks, Hedge P&L

Source: _bs-greeks-pnl.py

import math
import random

# ══════════════════════════════════════════════════════════════════════════════
# BLACK-SCHOLES CALL & PUT PRICER
# ══════════════════════════════════════════════════════════════════════════════
# Assumptions: European option, no dividends, constant vol, log-normal returns
# S = spot price, K = strike, T = time to expiry (years), r = risk-free rate, sigma = vol
#
# d1 = [ln(S/K) + (r + 0.5*sigma^2)*T] / (sigma * sqrt(T))
# d2 = d1 - sigma * sqrt(T)
# Call = S * N(d1) - K * e^(-rT) * N(d2)
# Put  = K * e^(-rT) * N(-d2) - S * N(-d1)   [put-call parity]

def normal_cdf(x):
    # Phi(x): cumulative normal distribution using math.erf (no scipy needed)
    return 0.5 * (1 + math.erf(x / math.sqrt(2)))

def black_scholes(S, K, T, r, sigma, option_type='call'):
    d1 = (math.log(S / K) + (r + 0.5 * sigma ** 2) * T) / (sigma * math.sqrt(T))
    d2 = d1 - sigma * math.sqrt(T)
    if option_type == 'call':
        return S * normal_cdf(d1) - K * math.exp(-r * T) * normal_cdf(d2)
    elif option_type == 'put':
        return K * math.exp(-r * T) * normal_cdf(-d2) - S * normal_cdf(-d1)

# Example: ATM call, 1 year, 20% vol, 5% risk-free
print(f"BS Call: {black_scholes(S=100, K=100, T=1, r=0.05, sigma=0.20):.4f}")
print(f"BS Put:  {black_scholes(S=100, K=100, T=1, r=0.05, sigma=0.20, option_type='put'):.4f}")
# Call > Put for same strike when r > 0 (carry effect)


# ══════════════════════════════════════════════════════════════════════════════
# GREEKS (from Black-Scholes)
# ══════════════════════════════════════════════════════════════════════════════
# Delta: sensitivity of option price to <span class="nokatex">$</span>1 move in spot
# Gamma: sensitivity of delta to <span class="nokatex">$</span>1 move in spot (risk manager cares: hedge slippage)
# Vega:  sensitivity to 1 point (1%) move in vol — reported per 1% vol move
# Theta: daily time decay (negative for long options)

def normal_pdf(x):
    # standard normal density — needed for gamma and vega
    return math.exp(-0.5 * x ** 2) / math.sqrt(2 * math.pi)

def greeks(S, K, T, r, sigma, option_type='call'):
    d1 = (math.log(S / K) + (r + 0.5 * sigma ** 2) * T) / (sigma * math.sqrt(T))
    d2 = d1 - sigma * math.sqrt(T)

    if option_type == 'call':
        delta = normal_cdf(d1)
    else:
        delta = normal_cdf(d1) - 1          # put delta is negative

    gamma = normal_pdf(d1) / (S * sigma * math.sqrt(T))  # same for call and put
    vega  = S * normal_pdf(d1) * math.sqrt(T) / 100      # per 1% vol move
    if option_type == 'call':
        theta = (- S * normal_pdf(d1) * sigma / (2 * math.sqrt(T))
                 - r * K * math.exp(-r * T) * normal_cdf(d2)) / 365
    else:
        theta = (- S * normal_pdf(d1) * sigma / (2 * math.sqrt(T))
                 + r * K * math.exp(-r * T) * normal_cdf(-d2)) / 365

    return {'delta': delta, 'gamma': gamma, 'vega': vega, 'theta': theta}

g = greeks(S=100, K=100, T=1, r=0.05, sigma=0.20)
for k, v in g.items():
    print(f"{k:6s}: {v:.6f}")


# ══════════════════════════════════════════════════════════════════════════════
# DELTA HEDGE P&L
# ══════════════════════════════════════════════════════════════════════════════
# You sold a call, delta-hedged by buying delta shares.
# If spot moves, your hedge P&L offsets option P&L — but gamma causes slippage.
# Risk manager monitors: hedge ratio drift, gamma exposure, rehedge frequency.

def delta_hedge_pnl(S0, S1, K, T, r, sigma):
    # position: short 1 call, long delta shares at S0
    call0  = black_scholes(S0, K, T,       r, sigma, 'call')
    call1  = black_scholes(S1, K, T - 1/252, r, sigma, 'call')  # 1 day later
    delta0 = greeks(S0, K, T, r, sigma)['delta']

    option_pnl = -(call1 - call0)   # short call: profit if price falls
    hedge_pnl  = delta0 * (S1 - S0) # long delta shares
    return option_pnl + hedge_pnl   # residual = gamma P&L

print(f"\nDelta hedge P&L (spot +5): {delta_hedge_pnl(100, 105, 100, 1, 0.05, 0.20):.4f}")
print(f"Delta hedge P&L (spot -5): {delta_hedge_pnl(100, 95,  100, 1, 0.05, 0.20):.4f}")
# Both negative because short gamma: you lose when spot moves either way


# ══════════════════════════════════════════════════════════════════════════════
# PUT-CALL PARITY CHECK
# ══════════════════════════════════════════════════════════════════════════════
# C - P = S - K * e^(-rT)   (for European options, no dividends)
# If this doesn't hold -> arbitrage opportunity exists

def check_put_call_parity(S, K, T, r, sigma):
    C = black_scholes(S, K, T, r, sigma, 'call')
    P = black_scholes(S, K, T, r, sigma, 'put')
    lhs = C - P
    rhs = S - K * math.exp(-r * T)
    print(f"C - P = {lhs:.4f} | S - PV(K) = {rhs:.4f} | diff = {abs(lhs-rhs):.10f}")

check_put_call_parity(100, 100, 1, 0.05, 0.20)


# ══════════════════════════════════════════════════════════════════════════════
# OPTION PAYOFF AT EXPIRY (no model needed)
# ══════════════════════════════════════════════════════════════════════════════
# Net P&L = intrinsic value at expiry - premium paid

def option_payoff(spot_at_expiry, strike, premium, option_type='call', position='long'):
    if option_type == 'call':
        intrinsic = max(spot_at_expiry - strike, 0)
    else:
        intrinsic = max(strike - spot_at_expiry, 0)
    pnl = intrinsic - premium
    return pnl if position == 'long' else -pnl

# Long call: paid <span class="nokatex">$</span>5 premium, strike 100
for spot in [85, 95, 100, 105, 115]:
    pnl = option_payoff(spot, strike=100, premium=5, option_type='call', position='long')
    print(f"Spot={spot:3d} | Long Call P&L: {pnl:+.2f}")


# ══════════════════════════════════════════════════════════════════════════════
# EXPECTED VALUE — PROBABILITY SIMULATION
# ══════════════════════════════════════════════════════════════════════════════
# Classic Akuna/trading interview: "simulate this game and find the EV"
# Example: roll a die, get paid the face value. What's the EV?
# Analytical: (1+2+3+4+5+6)/6 = 3.5

def simulate_ev(n=100000):
    total = sum(random.randint(1, 6) for _ in range(n))
    return total / n

print(f"\nSimulated die EV: {simulate_ev():.4f}")  # should be ~3.5

# Variant: roll twice, take the max. What's the EV?
def simulate_max_of_two(n=100000):
    total = sum(max(random.randint(1, 6), random.randint(1, 6)) for _ in range(n))
    return total / n

print(f"EV of max(d1, d2): {simulate_max_of_two():.4f}")  # analytical: 161/36 ≈ 4.472


# ══════════════════════════════════════════════════════════════════════════════
# EXPECTED TOSSES UNTIL 2 HEADS IN A ROW
# ══════════════════════════════════════════════════════════════════════════════
# Classic brain teaser — analytical answer is 6. Simulate to verify.

def simulate_two_heads(n=100000):
    total_tosses = 0
    for _ in range(n):
        tosses = 0
        consecutive = 0
        while consecutive < 2:
            tosses += 1
            if random.random() < 0.5:   # heads
                consecutive += 1
            else:
                consecutive = 0
        total_tosses += tosses
    return total_tosses / n

print(f"Expected tosses for 2 heads in a row: {simulate_two_heads():.4f}")  # ~6.0
